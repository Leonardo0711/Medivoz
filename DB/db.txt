Project medivoz_vps_postgresql {
  database_type: "PostgreSQL"
  Note: '''
  Esquema completo para una plataforma de consulta medica asistida por IA en tiempo real,
  desplegada sobre PostgreSQL propio en un VPS.

  Decisiones de esta version:
  1) Registro simple con correo, nombre completo, especialidad y contrasena.
  2) No hay verificacion de correo ni MFA en esta etapa.
  3) El administrador existe, pero se crea manualmente.
  4) Los usuarios que se registran desde la app se crean como doctores.
  5) Fase 1 se enfoca en anamnesis de primera consulta, no en historia clinica longitudinal.
  6) Se mantiene la tabla pacientes para distinguir pacientes entre consultas, pero con datos minimos.
  7) Los audios SI pueden existir, pero solo de forma temporal.
  8) El audio se usa para procesamiento y depuracion corta, luego se elimina.
  9) La consulta es la entidad central.
  10) La transcripcion se guarda por segmentos para permitir llenado en vivo.
  11) La ficha medica se llena por secciones de anamnesis.
  12) La estructura de anamnesis se prepara por especialidad, aunque inicialmente todas compartan los mismos campos.
  13) Los agentes se definen por plantilla y se asignan a cada doctor.
  14) Mas adelante el doctor puede personalizar configuracion y prompt del agente.
  15) Se agrega manejo robusto de sesiones con refresh token por dispositivo.
  '''
}

Table usuarios {
  id uuid [pk, default: `gen_random_uuid()`]
  correo_electronico varchar(255) [not null]
  hash_contrasena text [not null, note: "Nunca guardar la contrasena en texto plano"]
  estado_cuenta estado_cuenta [not null, default: 'activa']
  ultimo_acceso_en timestamptz
  creado_en timestamptz [not null, default: `now()`]
  actualizado_en timestamptz [not null, default: `now()`]

  Indexes {
    correo_electronico [unique]
    estado_cuenta
  }

  Note: '''
  Tabla principal de autenticacion.
  El usuario se registra con correo y contrasena.
  No hay verificacion de correo en esta version.
  El administrador principal se crea manualmente.
  '''
}

Table sesiones_usuario {
  id uuid [pk, default: `gen_random_uuid()`]
  usuario_id uuid [not null, ref: > usuarios.id]
  refresh_token_hash text [not null, note: "Guardar hash del refresh token, no el token plano"]
  ip_origen varchar(45)
  agente_usuario text [note: "User-Agent del navegador o app"]
  nombre_dispositivo text [note: "Opcional: Chrome Windows, iPhone, Android, etc."]
  expira_en timestamptz [not null]
  ultimo_uso_en timestamptz
  revocada_en timestamptz
  creado_en timestamptz [not null, default: `now()`]
  actualizado_en timestamptz [not null, default: `now()`]

  Indexes {
    refresh_token_hash [unique]
    usuario_id
    expira_en
    revocada_en
  }

  Note: '''
  Representa una sesion real de autenticacion por dispositivo o navegador.
  Permite manejar refresh tokens, cierre de sesion, revocacion e invalidacion
  sin guardar tokens planos en la base de datos.
  '''
}

}

Table pacientes {
  id uuid [pk, default: `gen_random_uuid()`]
  doctor_id uuid [not null, ref: > usuarios.id]
  codigo_paciente varchar(40) [not null, default: `('PAC-' || upper(substr(gen_random_uuid()::text, 1, 8)))`, note: "Codigo interno para diferenciar pacientes aunque no se registre DNI"]
  nombres_apellidos text [not null]
  dni varchar(20)
  edad int
  ocupacion text
  procedencia text
  diagnostico_general text
  ultima_visita timestamptz
  creado_en timestamptz [not null, default: `now()`]
  actualizado_en timestamptz [not null, default: `now()`]

  Indexes {
    (doctor_id, codigo_paciente) [unique]
    (doctor_id, dni) [unique]
    doctor_id
    nombres_apellidos
  }

  Note: '''
  Catalogo minimo de pacientes de cada doctor para fase 1.
  Se mantiene para distinguir pacientes entre consultas sin exigir un perfil clinico longitudinal.
  codigo_paciente permite identificar registros cuando no se tiene DNI.
  ultima_visita es un dato derivado o cache de la ultima consulta finalizada.
  Si no existen consultas previas, puede quedar nulo.
  '''
}

Table consultas {
  id uuid [pk, default: `gen_random_uuid()`]
  doctor_id uuid [not null, ref: > usuarios.id]
  paciente_id uuid [not null, ref: > pacientes.id]
  plantilla_anamnesis_id uuid [ref: > plantillas_anamnesis.id]
  codigo_consulta varchar(40) [not null]
  tipo_consulta tipo_consulta [not null, default: 'primera_consulta']
  estado estado_consulta [not null, default: 'en_espera']
  fecha_hora_consulta timestamptz
  inicio_real timestamptz
  fin_real timestamptz

  transcripcion_completa text [note: "Texto consolidado generado a partir de los segmentos"]
  version_transcripcion int [not null, default: 1]

  creado_en timestamptz [not null, default: `now()`]
  actualizado_en timestamptz [not null, default: `now()`]

  Indexes {
    codigo_consulta [unique]
    doctor_id
    paciente_id
    plantilla_anamnesis_id
    tipo_consulta
    estado
    fecha_hora_consulta
  }

  Note: '''
  Entidad central del dominio clinico.
  Representa una primera consulta de anamnesis en fase 1.
  tipo_consulta queda listo para controles o seguimiento futuros, aunque inicialmente sea primera_consulta.
  Desde aqui se relacionan segmentos, audio temporal, ejecuciones y ficha medica.
  '''
}

Table audios_temporales_consulta {
  id uuid [pk, default: `gen_random_uuid()`]
  consulta_id uuid [not null, ref: > consultas.id]
  ruta_almacenamiento text [not null]
  tipo_mime varchar(100)
  duracion_ms int
  tamano_bytes bigint
  hash_archivo text
  estado estado_audio_temporal [not null, default: 'pendiente_procesamiento']
  motivo_conservacion text [note: "Ejemplo: procesamiento, depuracion, reintento, revision tecnica"]
  expira_en timestamptz [not null]
  borrado_en timestamptz
  creado_en timestamptz [not null, default: `now()`]
  actualizado_en timestamptz [not null, default: `now()`]

  Indexes {
    consulta_id
    estado
    expira_en
  }

  Note: '''
  Audio temporal de una consulta.
  No es almacenamiento permanente.
  Se usa solo para procesamiento o depuracion corta y luego debe eliminarse automaticamente.
  Un worker o tarea programada deberia borrar todo audio vencido.
  '''
}

Table segmentos_transcripcion {
  id uuid [pk, default: `gen_random_uuid()`]
  consulta_id uuid [not null, ref: > consultas.id]
  numero_secuencia int [not null]
  hablante tipo_hablante [not null, default: 'desconocido']
  origen origen_transcripcion [not null, default: 'flujo_en_vivo']

  inicio_ms int
  fin_ms int
  texto text [not null]
  confianza decimal(5,4)
  codigo_idioma varchar(10)
  carga_cruda jsonb

  creado_en timestamptz [not null, default: `now()`]
  actualizado_en timestamptz [not null, default: `now()`]

  Indexes {
    (consulta_id, numero_secuencia) [unique]
    consulta_id
    hablante
  }

  Note: '''
  Transcripcion incremental de la consulta.
  Cada fila representa un segmento pequeno ya convertido a texto.
  Esto permite que la IA empiece a llenar la ficha antes de terminar toda la consulta.
  '''
}
